``` c++
#include <bits/stdc++.h>
using namespace std;

const int N = 4e5 + 10, inf = 0x3f3f3f3f;
int n, m;
int dis[N], num[N], pre[N], inq[N], mark[N];
struct edge {
    int u, v, c, f, w;
};
vector<edge> edges;
vector<int> G[N];
queue<int> q;

void add(int u, int v, int c) {
    edges.push_back((edge){u, v, 1, 0, c});
    edges.push_back((edge){v, u, 0, 0, -c});
    int tmp = edges.size();
    G[u].push_back(tmp - 2);
    G[v].push_back(tmp - 1);
}

bool spfa(int s, int t, int &flow, int &cost) {
    memset(dis, 0x3f, sizeof(dis));
    memset(pre, 0, sizeof(pre));
    memset(inq, 0, sizeof(inq));
    dis[s] = 0, inq[s] = 1, num[s] = inf, pre[s] = 0;
    q.push(s);
    while (!q.empty()) {
        int u = q.front(); q.pop(); inq[u] = 0;
        for (int i = 0; i < G[u].size(); i++) {
            edge x = edges[G[u][i]];
            if (x.c > x.f && dis[x.v] > dis[u] + x.w) {
                dis[x.v] = dis[u] + x.w;
                pre[x.v] = G[u][i];
                num[x.v] = min(num[u], x.c - x.f);
                if (!inq[x.v]) inq[x.v] = 1, q.push(x.v);
            }
        }
    }
    if (dis[t] == inf) return 0;
    flow += num[t];
    cost += dis[t] * num[t];
    int u = t;
    while (u != s) {
        edges[pre[u]].f += num[t], edges[pre[u] ^ 1].f -= num[t];
        u = edges[pre[u]].u;
    }
    return 1;
}

void Mincost(int s, int t) {
    int flow = 0, cost = 0;
    while (spfa(s, t, flow, cost));
    printf("%d %d\n", flow, cost);
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 2; i < n; i++) add(i, i + n, 0);
    for (int i = 1; i <= m; i++) {
        int a, b, c; scanf("%d%d%d", &a, &b, &c);
        add(a + n, b, c);
    }
    Mincost(1 + n, n);
    return 0;
}
```
