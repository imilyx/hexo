[题目链接](https://www.luogu.org/problemnew/show/P3258)

唯一需要注意的是，除起点外每一个点都会计算两次（多算一次），所以要减回来。

code:
``` c++
#include <bits/stdc++.h>
using namespace std;

const int N = 3e5 + 10;
int n;
int a[N], f[N][30], dep[N], pre[N], sum[N];
int to[N << 1], nxt[N << 1], lnk[N], cnt;

void add(int x, int y) {
    to[++cnt] = y, nxt[cnt] = lnk[x], lnk[x] = cnt;
}

void dfs(int x, int fa) {
    for (int i = 1; i <= 28; i++)
        f[x][i] = f[f[x][i - 1]][i - 1];
    for (int i = lnk[x]; i; i = nxt[i]) {
        int y = to[i];
        if (y == fa) continue;
        dep[y] = dep[x] + 1;
        f[y][0] = x;
        dfs(y, x);
    }
}

int lca(int x, int y) {
    if (dep[x] < dep[y]) swap(x, y);
    for (int i = 28; i >= 0; i--)
        if (dep[f[x][i]] >= dep[y]) x = f[x][i];
    if (x == y) return x;
    for (int i = 28; i >= 0; i--)
        if (f[x][i] != f[y][i]) x = f[x][i], y = f[y][i];
    return f[x][0];
}

void upd(int x, int fa) {
    for (int i = lnk[x]; i; i = nxt[i]) {
        int y = to[i];
        if (y == fa) continue;
        upd(y, x);
        pre[x] += pre[y];
    }
}

int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for (int i = 1; i < n; i++) {
        int x, y; scanf("%d%d", &x, &y);
        add(x, y), add(y, x);
    }
    dep[1] = 1;
    dfs(1, 0);
    for (int i = 1; i < n; i++) {
        int xy = lca(a[i], a[i + 1]);
        pre[a[i]]++, pre[a[i + 1]]++, pre[f[xy][0]]--, pre[xy]--;
    }
    upd(1, 0);
    for (int i = 2; i <= n; i++) pre[a[i]]--;  // 减回来！！
    for (int i = 1; i <= n; i++) printf("%d\n", pre[i]);
    return 0;
}
```
