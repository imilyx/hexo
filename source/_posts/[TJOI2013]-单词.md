[传送门](https://www.luogu.org/problem/P3966)

显然，如果每个字符都不一样，建好 AC 自动机后每个串在自动机上跑一遍，显然 T 得飞起。。

怎么办呢？好东西来了：Fail 树。顾名思义，就是所有 Fail 指针反向构成的树。

性质：在 Fail 树中，若 y 在 x 的子树中，则 x 是 y 的子串。

code :
``` c++
#include <bits/stdc++.h>
#define rep(i, x, y) for (int i = x; i <= y; i++)
using namespace std;

const int N = 1e6 + 10;
char s[N];
int n, idx[N], ch[N][26], sz, f[N], cnt[N];
vector<int> G[N];

int ord(char c) { return c - 'a'; }

int insert(char s[]) {
    int u = 0, n = strlen(s + 1);
    rep(i, 1, n) {
        int c = ord(s[i]);
        if (!ch[u][c]) ch[u][c] = ++sz;
        u = ch[u][c];
        cnt[u]++;
    }
    return u;
}

void getfail() {
    queue<int> q;
    f[0] = 0;
    for (int c = 0; c < 26; c++) {
        int u = ch[0][c];
        if (!u) continue;
        f[u] = 0;
        q.push(u);
        G[0].push_back(u);
    }
    while (q.size()) {
        int x = q.front(); q.pop();
        for (int c = 0; c < 26; c++) {
            int u = ch[x][c];
            if (!u) continue;
            q.push(u);
            int y = f[x];
            while (y && !ch[y][c]) y = f[y];
            f[u] = ch[y][c];
            G[f[u]].push_back(u);
        }
    }
}

void dfs(int x, int fa) {
    for (int i = 0; i < G[x].size(); i++) {
        int y = G[x][i];
        if (y == fa) continue;
        dfs(y, x);
        cnt[x] += cnt[y];
    }
}

int main() {
    cin >> n;
    rep(i, 1, n) {
        scanf("%s", s + 1);
        idx[i] = insert(s);
    }
    getfail();
    dfs(0, -1);
    rep(i, 1, n) printf("%d\n", cnt[idx[i]]);
    return 0;
}
```
